135  // Let's setup the <get> end-point.
136  this->Get(route, this->Authorize([this, route, path](const Request& req,
137  Response& res) {
138  // Let's check if given uri path is starting from the base one..
139  auto base_path = std::filesystem::path(path);
140  auto base_path_absolute = std::filesystem::absolute(base_path);
141  auto route_filtered = req.Uri.GetPath();
142  auto path_pos = route_filtered.find(route);
143  if (!path_pos) {
144  route_filtered.erase(path_pos, route.length());
145  }
146  auto uri_path_absolute = std::filesystem::absolute(
147  base_path.concat(std::filesystem::path(route_filtered).string()));
148  auto uri_string = uri_path_absolute.string();
149  if (!uri_string.find(base_path_absolute.string())) {
150  // Let's try to retrieve the required content..
151  auto content = get(uri_string);
152  if (content.first.has_value()) {
153  ResponseBuilder::Build<response_writers::TransferEncoding>(
154  res, content.first.value(), content.second)
155  .Ok_200();

187  // Retrieves (if possible) the required resource (using cache).
188  std::pair<std::optional<base::AutoBuffer>, std::string> get(
189  const std::string& key) {
190  auto resource = cache_.Get(key);
191  if (!resource.has_value()) {
192  resource = repository_.Get(key);
193  }
194  return std::make_pair(resource, getMimeType(key));

196  // Tries to extract the mime-type from the resource name.
197  std::string getMimeType(const std::string& resource_name) {
198  std::string mime_type = constants::Mime::kBIN;
199  auto ex = std::filesystem::path(resource_name).extension().string();
200  std::transform(ex.begin(), ex.end(), ex.begin(), ::tolower);
201  auto itr = extensions_.find(ex);
202  if (itr != extensions_.end()) {
203  mime_type = itr->second;
204  }
205  return mime_type;
206  }
