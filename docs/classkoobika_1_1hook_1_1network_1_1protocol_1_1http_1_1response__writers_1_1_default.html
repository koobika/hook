Example

int main() {
try {
auto server = ServerBuilder().Build();
server->Handle("/foo/bar", [](const Request& req, Response& res) {
if (req.Method.IsGet()) {
ResponseBuilder::Build<response_writers::TransferEncoding>(
res, json::Array{"Hello, ", "World!"}, constants::Mime::kTXT)
.Ok_200();
} else {
ResponseBuilder::Build<response_writers::Default>(
res, "Requested method is not supported!", constants::Mime::kTXT)
.Forbidden_403();
}
});
server->Start("8080");
return getchar();
} catch (const std::exception& exception) {
// ((Error)) -> while performing setup!
std::cout << exception.what() << std::endl;
return -1;
}
}
