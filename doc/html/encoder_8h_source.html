48  public:

52  // Returns the encoded content (buffer-based).
53  static std::string Encode(
54  const std::string& in,
55  const EncoderType& encoder_type = EncoderType::Base64) {
56  std::string encoded;
57  const char* alphabet = encoder_type == EncoderType::Base64
58  ? Constants::kBase64EncodingAlphabet
59  : Constants::kBase64UrlEncodingAlphabet;
60  // check if input data is less than 24-bit
61  bool byte_added = ((in.length() * 8) % 6);
62  // encode input data
63  const char* input_buffer_str = (const char*)in.data();
64  std::size_t input_buffer_str_ori_len = in.length();
65  std::size_t input_buffer_str_len = input_buffer_str_ori_len + byte_added;
66  unsigned char r = 0;
67  static constexpr unsigned char c_disps[] = {0x02, 0x04, 0x06};
68  static constexpr unsigned char r_disps[] = {0x04, 0x02, 0x00};
69  static constexpr unsigned char c_masks[] = {0xFC, 0xF0, 0xC0};
70  static constexpr unsigned char r_masks[] = {0x03, 0x0F, 0x3F};
71  std::size_t elements = sizeof(c_masks);
72  for (std::size_t i = 0; i < input_buffer_str_len; i++) {
73  bool last_element = (i == (input_buffer_str_len - 1));
74  bool last_loop_element = (i && (i % elements) == 0);
75  if (last_loop_element) {
76  encoded += alphabet[r];
77  r = 0;
78  }
79  unsigned char val =
80  last_element ? byte_added ? Constants::kZeroCh : input_buffer_str[i]
81  : input_buffer_str[i];
82  auto this_byte_mask = c_masks[i % elements];
83  auto remainder_mask = r_masks[i % elements];
84  auto index = ((val & this_byte_mask) >> c_disps[i % elements]) | r;
85  int r_i = (val & remainder_mask) << r_disps[i % elements];
86  r = (unsigned char)r_i;
87  encoded += alphabet[index];
88  }
89  if (r) encoded += alphabet[r];
90  // check for final padding..
91  if (input_buffer_str_ori_len % 3) {
92  while (encoded.length() % 4) {
93  encoded += Constants::kPadCh;
94  }
95  }
96  return encoded;
97  }
98 };
