87  // ___________________________________________________________________________
88  // METHODs ( public )
89  //
90  // Returns the stored value (if any) using the provided type specifier.
91  template <typename OUty>
92  std::optional<OUty> Get() const {
93  auto ptr = std::get_if<OUty>(&data_);
94  return ptr != nullptr ? OUty(*ptr) : std::optional<OUty>{};
95  }
96  // Returns the stored value (if any) as string.
97  std::optional<std::string> GetAsString() const {
98  switch (data_.index()) {
99  case kBool_:
100  return std::get<kBool_>(data_) ? "true" : "false";
101  case kUnsignedChar_:
102  return std::to_string(std::get<kUnsignedChar_>(data_));
103  case kChar_:
104  return std::to_string(std::get<kChar_>(data_));
105  case kUnsignedShort_:
106  return std::to_string(std::get<kUnsignedShort_>(data_));
107  case kShort_:
108  return std::to_string(std::get<kShort_>(data_));
109  case kUnsignedInt_:
110  return std::to_string(std::get<kUnsignedInt_>(data_));
111  case kInt_:
112  return std::to_string(std::get<kInt_>(data_));
113  case kUnsignedLong_:
114  return std::to_string(std::get<kUnsignedLong_>(data_));
115  case kLong_:
116  return std::to_string(std::get<kLong_>(data_));
117  case kFloat_:
118  return std::to_string(std::get<kFloat_>(data_));
119  case kDouble_:
120  return std::to_string(std::get<kDouble_>(data_));
121  case kString_:
122  return std::get<kString_>(data_);
123  }
124  return {};
125  }
126 
127  private:
