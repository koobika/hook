61 class Basic : public Controller<ContextBasic> {
62  public:
63  // ___________________________________________________________________________
64  // CONSTRUCTORs/DESTRUCTORs ( public )
65  //
66  Basic(const Checker& user_checker = nullptr,
67  const double session_timeout_in_seconds = kDefaultTimeout)
68  : timeout_in_seconds_{session_timeout_in_seconds} {
69  this->Check = (user_checker != nullptr)
70  ? user_checker
71  : [this](const Context& context) -> bool {
72  do {
73  auto itr = data_.find(context.Username);
74  if (itr == data_.end() || itr->second.first != context.Password) break;
75  if (timeout_in_seconds_ > 0.0) {
76  auto current = std::chrono::system_clock().now();
77  std::chrono::duration<double> diff = current - itr->second.second;
78  if (diff.count() >= timeout_in_seconds_) break;
79  }
80  return true;
81  } while (false);
82  return false;
83  };
84  }
85  Basic(const Basic&) = default;
86  Basic(Basic&&) noexcept = default;
87  ~Basic() = default;
88  // ___________________________________________________________________________
89  // OPERATORs ( public )
90  //
91  Basic& operator=(const Basic&) = default;
92  Basic& operator=(Basic&&) noexcept = default;
93  // ___________________________________________________________________________
94  // METHODs ( public )
95  //
96  // Sets user/password tuple within the database.
97  void Set(const std::string& user, const std::string& password) {
98  data_[user] = std::make_pair(password, std::chrono::system_clock().now());
99  }
100  // Clears specific user from database.
101  void Clear(const std::string& user) { data_.erase(user); }
102  // Clears entire user/password database.
103  void Clear() { data_.clear(); }
104 
105  protected:
106  // ___________________________________________________________________________
107  // CONSTANTs ( protected )
108  //
109  static constexpr double kDefaultTimeout = -1.0; // disabled by default
110  // ___________________________________________________________________________
111  // ATTRIBUTEs ( protected )
112  //
113  std::unordered_map<
114  std::string,
115  std::pair<std::string, std::chrono::system_clock::time_point>>
116  data_;
117  double timeout_in_seconds_;
118 };
