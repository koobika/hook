83  void From(const char* str, const std::size_t& len) {
84  auto ptr = str;
85  auto end = ptr + len;
86  auto last = ptr;
87  bool field_name_ok = false;
88  const char* colon_delimiter = nullptr;
89  while (ptr != end) {
90  if (field_name_ok) {
91  if ((ptr + 1) != end && *ptr == constants::Characters::kCr &&
92  *(ptr + 1) == constants::Characters::kLf) {
93  if ((ptr + 2) == end ||
94  (*(ptr + 2) != constants::Characters::kSpace &&
95  *(ptr + 2) != constants::Characters::kHt)) {
96  while (*colon_delimiter == constants::Characters::kColon ||
97  *colon_delimiter == constants::Characters::kSpace ||
98  *colon_delimiter == constants::Characters::kHt)
99  colon_delimiter++;
100  Set(std::string(last, colon_delimiter - last - 2),
101  std::string(colon_delimiter, ptr - colon_delimiter));
102  ptr += 2;
103  last = ptr;
104  field_name_ok = false;
105  continue;
106  }
107  }
108  } else {
109  if ((field_name_ok = *ptr == constants::Characters::kColon)) {
110  colon_delimiter = ptr;
111  }
112  }
113  ptr++;
114  }
115  }
