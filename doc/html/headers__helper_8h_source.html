78  // Sets headers content from specified string.
79  void From(const char* str, const std::size_t& len) {
80  auto ptr = str;
81  auto end = ptr + len;
82  auto last = ptr;
83  bool field_name_ok = false;
84  const char* colon_delimiter = nullptr;
85  while (ptr != end) {
86  if (field_name_ok) {
87  if ((ptr + 1) != end && *ptr == constants::Characters::kCr &&
88  *(ptr + 1) == constants::Characters::kLf) {
89  if ((ptr + 2) == end ||
90  (*(ptr + 2) != constants::Characters::kSpace &&
91  *(ptr + 2) != constants::Characters::kHt)) {
92  while (*colon_delimiter == constants::Characters::kColon ||
93  *colon_delimiter == constants::Characters::kSpace ||
94  *colon_delimiter == constants::Characters::kHt)
95  colon_delimiter++;
96  Set(std::string(last, colon_delimiter - last - 2),
97  std::string(colon_delimiter, ptr - colon_delimiter));
98  ptr += 2;
99  last = ptr;
100  field_name_ok = false;
101  continue;
102  }
103  }
104  } else {
105  if ((field_name_ok = *ptr == constants::Characters::kColon)) {
106  colon_delimiter = ptr;
107  }
108  }
109  ptr++;
110  }
111  }
