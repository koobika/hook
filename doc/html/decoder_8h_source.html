55  // Returns the encoded content.
56  static std::string Decode(const std::string& in) {
57  static const std::unordered_map<char, char> map = {
58  {'A', 0x00}, {'B', 0x01}, {'C', 0x02}, {'D', 0x03}, {'E', 0x04},
59  {'F', 0x05}, {'G', 0x06}, {'H', 0x07}, {'I', 0x08}, {'J', 0x09},
60  {'K', 0x0A}, {'L', 0x0B}, {'M', 0x0C}, {'N', 0x0D}, {'O', 0x0E},
61  {'P', 0x0F}, {'Q', 0x10}, {'R', 0x11}, {'S', 0x12}, {'T', 0x13},
62  {'U', 0x14}, {'V', 0x15}, {'W', 0x16}, {'X', 0x17}, {'Y', 0x18},
63  {'Z', 0x19}, {'a', 0x1A}, {'b', 0x1B}, {'c', 0x1C}, {'d', 0x1D},
64  {'e', 0x1E}, {'f', 0x1F}, {'g', 0x20}, {'h', 0x21}, {'i', 0x22},
65  {'j', 0x23}, {'k', 0x24}, {'l', 0x25}, {'m', 0x26}, {'n', 0x27},
66  {'o', 0x28}, {'p', 0x29}, {'q', 0x2A}, {'r', 0x2B}, {'s', 0x2C},
67  {'t', 0x2D}, {'u', 0x2E}, {'v', 0x2F}, {'w', 0x30}, {'x', 0x31},
68  {'y', 0x32}, {'z', 0x33}, {'0', 0x34}, {'1', 0x35}, {'2', 0x36},
69  {'3', 0x37}, {'4', 0x38}, {'5', 0x39}, {'6', 0x3A}, {'7', 0x3B},
70  {'8', 0x3C}, {'9', 0x3D}, {'+', 0x3E}, {'/', 0x3F}, {'-', 0x3E},
71  {'_', 0x3F}};
72  std::string decoded;
73  auto input_length = in.length();
74  if (input_length % 4) {
75  // ((Error)) -> input data is not 24-bit multiple!
76  throw std::logic_error("invalid input data (not 24-bit multiple)!");
77  }
78  for (std::size_t i = 0; i < input_length; i += 4) {
79  auto pad = Constants::kPadCh;
80  // [first] byte extraction..
81  auto l_idx = i;
82  auto r_idx = i + 1;
83  auto l_itr = map.find(in[l_idx]);
84  auto r_itr = map.find(in[r_idx]);
85  if ((l_itr == map.end() && in[l_idx] != pad) ||
86  (r_itr == map.end() && in[r_idx] != pad)) {
87  // ((Error)) -> invalid base64 character!
88  throw std::logic_error("invalid base64 character found!");
89  }
90  auto l = in[l_idx] == pad ? pad : l_itr->second;
91  auto r = in[r_idx] == pad ? pad : r_itr->second;
92  int fB_i = ((l & 0b00111111) << 2) | ((r & 0b00110000) >> 4);
93  decoded += (char)fB_i;
94  // [second] byte extraction..
95  r_idx = i + 2;
96  if (in[r_idx] == Constants::kPadCh) break;
97  r_itr = map.find(in[r_idx]);
98  if (r_itr == map.end() && in[r_idx] != pad) {
99  // ((Error)) -> invalid base64 character!
100  throw std::logic_error("invalid base64 character found!");
101  }
102  l = r;
103  r = in[r_idx] == pad ? pad : r_itr->second;
104  int sB_i = ((l & 0b00001111) << 4) | ((r & 0b00111100) >> 2);
105  decoded += (char)sB_i;
106  // [third] byte extraction..
107  r_idx = i + 3;
108  if (in[r_idx] == Constants::kPadCh) break;
109  r_itr = map.find(in[r_idx]);
110  if (r_itr == map.end() && in[r_idx] != pad) {
111  // ((Error)) -> invalid base64 character!
112  throw std::logic_error("invalid base64 character found!");
113  }
114  l = r;
115  r = in[r_idx] == pad ? pad : r_itr->second;
116  int tB_i = ((l & 0b00000011) << 6) | (r & 0b00111111);
117  decoded += (char)tB_i;
118  }
119  return decoded;
120  }
121 };
