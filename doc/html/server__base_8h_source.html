101  void Start(const std::string& port) {
102  if (transport_ != nullptr) {
103  // ((Error)) -> server is already running!
104  throw std::logic_error("Server is already running!");
105  }
106  configuration_[transport::ServerConstants::kPortKey] = port;
107  // let's setup the required transport!
108  transport_ = std::make_unique<TRty>();
109  // let' start transport activity on a separate thread!
110  transport_thread_ = std::make_shared<std::thread>([this]() {
111  transport_->Start(configuration_, [this](auto const& req, auto& res,
112  auto const& sender) {
113  try {
114  if (!router_.Perform(req, res)) {
115  // ((Error)) -> route is not registered!
116  // ((To-Do)) -> inform user back?
117  res.NotFound_404();
118  }
119  } catch (std::exception& e) {
120  // ((Error)) -> a known exception was thrown!
121  // ((To-Do)) -> inform user back!
122  res.InternalServerError_500(e.what());
123  res.Body.Close();
124  } catch (...) {
125  // ((Error)) -> a unknown exception was thrown!
126  // ((To-Do)) -> inform user back!
127  res.InternalServerError_500("Unknown server exception!");
128  res.Body.Close();
129  }
130  // Check for a 'RAW' based stream!
131  if (!res.Raw.Length()) {
132  if (!res.Headers.Exist(constants::Headers::kTransferEncoding)) {
133  res.Headers.Set(constants::Headers::kContentLength,
134  res.Body.Length());
135  }
136  res.Raw.Write(constants::Strings::kHttpVersion)
137  .Write(constants::Strings::kSpace)
138  .Write(std::to_string(res.StatusCode))
139  .Write(constants::Strings::kSpace)
140  .Write(res.ReasonPhrase)
141  .Write(constants::Strings::kCrLf);
142  res.Headers.DumpTo(res.Raw);
143  // append <response-body>!
144  res.Raw.Write(res.Body);
145  }
146  sender(res.Raw);
147  });
148  });
149  // let's start the support thread!
150  support_thread_ = std::make_shared<std::thread>([this] {
151  using namespace std::chrono_literals;
152  while (transport_) {
153  updateInternalDate();
154  std::this_thread::sleep_for(1s);
155  }
156  });
157  }
159  void Stop() {
160  if (transport_ != nullptr) {
161  transport_->Stop();
162  }
163  if (transport_thread_ != nullptr && transport_thread_->joinable()) {
164  transport_thread_->join();
165  }
166  transport_ = nullptr;
167  if (support_thread_ != nullptr && support_thread_->joinable()) {
168  support_thread_->join();
169  }
170  }

186  }

188  internal->AddToRouter(router_);
189  }
